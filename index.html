function updateDisplay() { const xhr = new XMLHttpRequest(); xhr.open("GET", API_URL, true); xhr.onload = function () { try { if (xhr.status !== 200) throw new Error("HTTPエラー: " + xhr.status); const data = JSON.parse(xhr.responseText); const race = data[TARGET_RACE_INDEX]; if (!race) throw new Error("レースデータが見つかりません"); let html = `
🚤 ${race.raceName}
`; race.boats.forEach(boat => { html += `
${boat.no} ${boat.name} 倍率 ${boat.odds} 払戻 ¥${Number(boat.payout).toLocaleString()}
`; }); document.getElementById("display").innerHTML = html; } catch (err) { document.getElementById("display").innerHTML = `
⚠ パースエラー: ${err.message}
`; } }; xhr.onerror = function () { document.getElementById("display").innerHTML = `
⚠ 通信エラー（XHR失敗）
`; }; xhr.send(); }
